140  int i = (unsigned int)(s->_rx_buffer_head + 1) % SERIAL_BUFFER_SIZE;
141 
142  // if we should be storing the received character into the location
143  // just before the tail (meaning that the head would advance to the
144  // current location of the tail), we're about to overflow the buffer
145  // and so we don't write the character or advance the head.
146  if (i != s->_rx_buffer_tail) {
147  s->_rx_buffer[s->_rx_buffer_head] = c;

149  s->_rx_buffer_head = i;

305 bool EMSSerial::begin(unsigned long baud)
306 {
307  uint16_t baud_setting;
308 
309  *_ucsra = 0;
310 
311  _written = false;
312 
313 
314  baud_setting = (F_CPU / 8 / baud - 1) / 2;
315 
316  // assign the baud_setting, ubbr (USART Baud Rate Register) and UBRRL (eight least significant
317  // bits of the USART baud rate).
318  *_ubrrh = baud_setting >> 8;
319  *_ubrrl = baud_setting;
320 
321  // set the bits in ucsrb port enabling reception, transmission and complete interrupt.
322  sbi(*_ucsrb, _rxen);
323  sbi(*_ucsrb, _txen);
324  sbi(*_ucsrb, _rxcie);
325 }

413 size_t EMSSerial::write(uint8_t c)
414 {
415  _written = true;
416 
417  // UDRE Flag indicates if the transmit buffer (UDR) is ready to receive new data. If UDRE is
418  // one, the buffer is empty, and therefore ready to be written. Wait therefore until buffer
419  // empty signal.
420  while (!(bitRead(*_ucsra, UDRE0))) {}
421 
422  // write value in the trasmit buffer
423  *_udr = c;
424 
425  return 1;
426 }

435 void EMSSerial::writeEOF() {
436  uint8_t t;
437  cbi(*_ucsrb, _rxen); //disable reception
438  while (!(bitRead(*_ucsra, UDRE0))) {} // wait for data register empty
439  t = *_ucsrc; // save settings
440  sbi(*_ucsrc, UPM01); //set parity even
441  sbi(*_ucsra, TXC0); //reset TX-complete (seems to be needed to get parity change)
442  *_udr = 0; //directly write a break
443  while (!(bitRead(*_ucsra, TXC0))) {} //wait for transmit complete
444  *_ucsrc = t; //restore settings
445  sbi(*_ucsra, TXC0); //reset TX-complete (seems to be needed to get parity change)
446  sbi(*_ucsrb, _rxen); //re-enable reception
447 }
